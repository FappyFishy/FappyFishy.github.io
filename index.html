package card_game;

import java.util.Scanner;

public class Game {
    private int score;
    private Card currentCard;
    private Card nextCard;
    private final Deck deck;
    private final Scanner sc;

    public Game() {
        score = 0;
        deck = new Deck();
        sc = new Scanner(System.in);
        currentCard = deck.getNextCard();
    }

    public void play() {
        System.out.println("Welcome to the Card Game!");
        while (deck.areCardsLeft()) {
            System.out.println("Current card: " + currentCard);
            System.out.print("Higher or lower? (h/l): ");
            String guess = sc.nextLine();

            nextCard = deck.getNextCard();
            if (nextCard == null) {
                System.out.println("No more cards in the deck!");
                break;
            }

            System.out.println("Next card: " + nextCard);

            boolean correct = (guess.equals("h") && nextCard.getValue().getNumericalValue() > currentCard.getValue().getNumericalValue())
                    || (guess.equals("l") && nextCard.getValue().getNumericalValue() < currentCard.getValue().getNumericalValue());

            if (correct) {
                score++;
                System.out.println("Correct! Your score: " + score);
            } else {
                System.out.println("Wrong! Game over. Final score: " + score);
                break;
            }

            currentCard = nextCard;
        }
        System.out.println("Thanks for playing!");
    }
}
